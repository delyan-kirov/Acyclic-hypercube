# Function to delete isomorphic copies from a list. 

gap> RemoveIsoImages := function(GraphList, GraphImage)
local len, i, element;
i := 0;
len := Length(GraphList);
for element in GraphList do
i := i + 1;
if IsIsomorphicDigraph(element, GraphImage) then
Remove(GraphList, i);
fi;
od;
return GraphList;
